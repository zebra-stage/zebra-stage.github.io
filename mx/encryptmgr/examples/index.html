        <parm name="SdCardOperation" value="1"/>

Queries

Queries are not supported on Zebra devices running Android 11 or later.

    </characteristic>
</wap-provisioningdoc>

Output


<wap-provisioningdoc>
    <characteristic type="EncryptMgr" version="4.3">
        <parm name="SdCardOperation" value="1" />
    </characteristic>
</wap-provisioningdoc>
