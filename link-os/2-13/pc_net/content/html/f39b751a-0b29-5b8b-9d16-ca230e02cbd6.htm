Sends the appropriate reset command to the printer.

Remarks
You should call Close after this method, as resetting the printer will terminate the connection.
Examples
Copy
using System;
using Zebra.Sdk.Comm;
using Zebra.Sdk.Printer;

public class ToolsUtilExample {

    public static void Main(string[] args) {
        Connection connection = new TcpConnection("192.168.1.101", TcpConnection.DEFAULT_ZPL_TCP_PORT);
        try {
            connection.Open();
            ZebraPrinter printer = ZebraPrinterFactory.GetInstance(connection);
            printer.Reset();
        } catch (ConnectionException e) {
            Console.WriteLine(e.ToString());
        } catch (ZebraPrinterLanguageUnknownException e) {
            Console.WriteLine(e.ToString());
        } finally {
            connection.Close();
        }
    }
}
